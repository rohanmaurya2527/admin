spark = SparkSession.builder.appName("DataProcessingRDD").getOrCreate()
# Load CSV into DataFrame
df_iris = spark.read.csv("iris.csv", header=True, inferSchema=True)
# Rename columns to remove dots
df_iris = df_iris.toDF('ID','Sepal_Length','Sepal_Width','Petal_Length','Petal_Width','Species')

# SQL queries
df_iris.createOrReplaceTempView("iris_table")
# Petal length filters
df_iris.filter(col('Petal_Length') < 2.5).show()

spark.sql("SELECT Species, COUNT(*) as Count FROM iris_table GROUP BY Species").show()

    SELECT Species, MIN(Petal_Length) as Min_Petal_Length,
           MAX(Petal_Length) as Max_Petal_Length,
           AVG(Petal_Length) as Avg_Petal_Length
    FROM iris_table
    GROUP BY Species
""").show()
